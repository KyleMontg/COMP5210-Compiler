	push rbp
	mov  rbp, rsp
	sub rsp, 16
	push r12
	push r13
	push r14
	push r15

	; Allocate registers for Parameters

	; x = 10
	mov r10, 10
	; y = 25
	mov r11, 25
	; i = 0
	mov r11, 0
	jmp .%L0
.%L0:
	; %t0 = i< 20
	mov r12, r11
	cmp r12, 20
	setl al
	movzx r12, al
	; if r12 goto .%L1 else .%L3
	cmp r12, 1
	je .%L1
	jmp .%L3
.%L1:
	; x = x++ 1
	mov r10, r10
	inc r10
	jmp .%L2
.%L2:
	; i = i++ 1
	mov r11, r11
	inc r11
	jmp .%L0
.%L3:
	; return r10
	mov rax, r10
.%L4:

	pop r15
	pop r14
	pop r13
	pop r12
	add rsp, 16
	pop rbp
	ret
